import torch
from transformers import AutoProcessor, AutoModelForVision2Seq
from pydantic import BaseModel, Field

class LineItem(BaseModel):
    label: str
    value: float = Field(..., description="amount in PLN")

class Statement(BaseModel):
    period: str
    currency: str = "PLN"
    items: list[LineItem]

class OnPremExtractor:
    """Local vision + language. No document leaves the network."""

    def __init__(self, weights, device="cuda"):
        self.processor = AutoProcessor.from_pretrained(weights)
        self.model = AutoModelForVision2Seq.from_pretrained(
            weights, torch_dtype=torch.bfloat16
        ).to(device).eval()

    @torch.inference_mode()
    def layout(self, page):
        x = self.processor(images=page, return_tensors="pt").to(self.model.device)
        out = self.model.generate(**x, max_new_tokens=1024)
        return self.processor.batch_decode(out, skip_special_tokens=True)[0]

    def extract(self, scan):
        blocks = [self.layout(p) for p in self.paginate(scan)]
        return Statement.model_validate(self.to_schema(blocks))
